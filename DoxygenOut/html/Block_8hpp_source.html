32 void Forward(typename TTypes<Type, 3>::ConstTensor x,
33 typename TTypes<Type, 3>::Tensor y) {
34 PROFILE_TRACE_FN("Block");
35
36 // x: [B, T, C], y: [B, T, C]
37 const int B = x.dimension(0); // batch size
38 const int T = x.dimension(1); // sequence length
39 const int C = x.dimension(2); // embedding dimensionality (n_embd)
40 CHECK_EQ(B, y.dimension(0));
41 CHECK_EQ(T, y.dimension(1));
42 CHECK_EQ(C, y.dimension(2));
43
44 ln1_y_->LazyAllocate(B * T * C);
45 ln1_mean_->LazyAllocate(B * T);
46 ln1_rstd_->LazyAllocate(B * T);
47 att_y_->LazyAllocate(B * T * C);
48 residual1_->LazyAllocate(B * T * C);
49 ln2_y_->LazyAllocate(B * T * C);
50 ln2_mean_->LazyAllocate(B * T);
51 ln2_rstd_->LazyAllocate(B * T);
52 mlp_y_->LazyAllocate(B * T * C);
53
54 // LN1
55 auto x_2d = MakeConstMatrix(x.data(), B * T, C);
56 auto ln1_y_2d = MakeMatrix(ln1_y_->data<Type>(), B * T, C);
57 auto ln1_mean_1d = MakeFlat(ln1_mean_->data<Type>(), B * T);
58 auto ln1_rstd_1d = MakeFlat(ln1_rstd_->data<Type>(), B * T);
59 ln1_->Forward(x_2d, ln1_y_2d, ln1_mean_1d, ln1_rstd_1d);
60
61 // Attention
62 auto ln1_y_3d = MakeConst3DTensor(ln1_y_2d.data(), B, T, C);
63 auto att_y_3d = Make3DTensor(att_y_->data<Type>(), B, T, C);
64 attn_->Forward(ln1_y_3d, att_y_3d);
65
66 // Residual
67 auto x_1d = MakeConstFlat(x.data(), B * T * C);
68 auto att_y_1d = MakeConstFlat(att_y_->data<Type>(), B * T * C);
69 auto residual1_1d = MakeFlat(residual1_->data<Type>(), residual1_->size());
70 nn::Residual::Forward(x_1d, att_y_1d, residual1_1d);
71
72 // LN2
73 auto ln2_y_2d = MakeMatrix(ln2_y_->data<Type>(), B * T, C);
74 auto ln2_y_2d_const = MakeConstMatrix(ln2_y_->data<Type>(), B * T, C);
75 auto ln2_mean_1d = MakeFlat(ln2_mean_->data<Type>(), B * T);
76 auto ln2_rstd_1d = MakeFlat(ln2_rstd_->data<Type>(), B * T);
77 auto residual1_2d = MakeConstMatrix(residual1_->data<Type>(), B * T, C);
78 ln2_->Forward(residual1_2d, ln2_y_2d, ln2_mean_1d, ln2_rstd_1d);
79
80 // MLP
81 auto mlp_y_2d = MakeMatrix(mlp_y_->data<Type>(), B * T, C);
82 mlp_->Forward(ln2_y_2d_const, mlp_y_2d);
83
84 // Residual
85 auto residual1_1d_const =
86 MakeConstFlat(residual1_->data<Type>(), residual1_->size());
87 auto mlp_y_1d = MakeConstFlat(mlp_y_->data<Type>(), B * T * C);
88 auto y_1d = MakeFlat(y.data(), y.size());
89 nn::Residual::Forward(residual1_1d_const, mlp_y_1d, y_1d);
90 }
