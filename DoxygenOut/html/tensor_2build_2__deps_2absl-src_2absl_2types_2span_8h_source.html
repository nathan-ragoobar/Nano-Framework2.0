17// span.h
18// -----------------------------------------------------------------------------
19//
20// This header file defines a `Span<T>` type for holding a reference to existing
21// array data. The `Span` object, much like the `absl::string_view` object,
22// does not own such data itself, and the data being referenced by the span must
23// outlive the span itself. Unlike `view` type references, a span can hold a
24// reference to mutable data (and can mutate it for underlying types of
25// non-const T.) A span provides a lightweight way to pass a reference to such
26// data.
27//
28// Additionally, this header file defines `MakeSpan()` and `MakeConstSpan()`
29// factory functions, for clearly creating spans of type `Span<T>` or read-only
30// `Span<const T>` when such types may be difficult to identify due to issues
31// with implicit conversion.
32//
33// The C++20 draft standard includes a `std::span` type. As of June 2020, the
34// differences between `absl::Span` and `std::span` are:
35// * `absl::Span` has `operator==` (which is likely a design bug,
36// per https://abseil.io/blog/20180531-regular-types)
37// * `absl::Span` has the factory functions `MakeSpan()` and
38// `MakeConstSpan()`
39// * bounds-checked access to `absl::Span` is accomplished with `at()`
40// * `absl::Span` has compiler-provided move and copy constructors and
41// assignment. This is due to them being specified as `constexpr`, but that
42// implies const in C++11.
43// * A read-only `absl::Span<const T>` can be implicitly constructed from an
44// initializer list.
45// * `absl::Span` has no `bytes()`, `size_bytes()`, `as_bytes()`, or
46// `as_writable_bytes()` methods
47// * `absl::Span` has no static extent template parameter, nor constructors
48// which exist only because of the static extent parameter.
49// * `absl::Span` has an explicit mutable-reference constructor
50//
51// For more information, see the class comments below.

103//------------------------------------------------------------------------------
104// Span
105//------------------------------------------------------------------------------
106//
107// A `Span` is an "array reference" type for holding a reference of contiguous
108// array data; the `Span` object does not and cannot own such data itself. A
109// span provides an easy way to provide overloads for anything operating on
110// contiguous sequences without needing to manage pointers and array lengths
111// manually.
112
113// A span is conceptually a pointer (ptr) and a length (size) into an already
114// existing array of contiguous memory; the array it represents references the
115// elements "ptr[0] .. ptr[size-1]". Passing a properly-constructed `Span`
116// instead of raw pointers avoids many issues related to index out of bounds
117// errors.
118//
119// Spans may also be constructed from containers holding contiguous sequences.
120// Such containers must supply `data()` and `size() const` methods (e.g
121// `std::vector<T>`, `absl::InlinedVector<T, N>`). All implicit conversions to
122// `absl::Span` from such containers will create spans of type `const T`;
123// spans which can mutate their values (of type `T`) must use explicit
124// constructors.
125//
126// A `Span<T>` is somewhat analogous to an `absl::string_view`, but for an array
127// of elements of type `T`, and unlike an `absl::string_view`, a span can hold a
128// reference to mutable data. A user of `Span` must ensure that the data being
129// pointed to outlives the `Span` itself.
130//
131// You can construct a `Span<T>` in several ways:
132//
133// * Explicitly from a reference to a container type
134// * Explicitly from a pointer and size
135// * Implicitly from a container type (but only for spans of type `const T`)
136// * Using the `MakeSpan()` or `MakeConstSpan()` factory functions.
137//
138// Examples:
139//
140// // Construct a Span explicitly from a container:
141// std::vector<int> v = {1, 2, 3, 4, 5};
142// auto span = absl::Span<const int>(v);
143//
144// // Construct a Span explicitly from a C-style array:
145// int a[5] = {1, 2, 3, 4, 5};
146// auto span = absl::Span<const int>(a);
147//
148// // Construct a Span implicitly from a container
149// void MyRoutine(absl::Span<const int> a) {
150// ...
151// }
152// std::vector v = {1,2,3,4,5};
153// MyRoutine(v) // convert to Span<const T>
154//
155// Note that `Span` objects, in addition to requiring that the memory they
156// point to remains alive, must also ensure that such memory does not get
157// reallocated. Therefore, to avoid undefined behavior, containers with
158// associated spans should not invoke operations that may reallocate memory
159// (such as resizing) or invalidate iterators into the container.
160//
161// One common use for a `Span` is when passing arguments to a routine that can
162// accept a variety of array types (e.g. a `std::vector`, `absl::InlinedVector`,
163// a C-style array, etc.). Instead of creating overloads for each case, you
164// can simply specify a `Span` as the argument to such a routine.
165//
166// Example:
167//
168// void MyRoutine(absl::Span<const int> a) {
169// ...
170// }
171//
172// std::vector v = {1,2,3,4,5};
173// MyRoutine(v);
174//
175// absl::InlinedVector<int, 4> my_inline_vector;
176// MyRoutine(my_inline_vector);
177//
178// // Explicit constructor from pointer,size
179// int* my_array = new int[10];
180// MyRoutine(absl::Span<const int>(my_array, 10));
181template <typename T>
182class ABSL_ATTRIBUTE_VIEW Span {
183 private:
184 // Used to determine whether a Span can be constructed from a container of
185 // type C.
186 template <typename C>
187 using EnableIfConvertibleFrom =
188 typename std::enable_if<span_internal::HasData<T, C>::value &&
189 span_internal::HasSize<C>::value>::type;
190
191 // Used to SFINAE-enable a function when the slice elements are const.
192 template <typename U>
193 using EnableIfValueIsConst =
194 typename std::enable_if<std::is_const<T>::value, U>::type;
195
196 // Used to SFINAE-enable a function when the slice elements are mutable.
197 template <typename U>
198 using EnableIfValueIsMutable =
199 typename std::enable_if<!std::is_const<T>::value, U>::type;
200
201 public:
202 using element_type = T;
203 using value_type = absl::remove_cv_t<T>;
204 // TODO(b/316099902) - pointer should be Nullable<T*>, but this makes it hard
205 // to recognize foreach loops as safe.
206 using pointer = T*;
207 using const_pointer = const T*;
208 using reference = T&;
209 using const_reference = const T&;
210 using iterator = pointer;
211 using const_iterator = const_pointer;
212 using reverse_iterator = std::reverse_iterator<iterator>;
213 using const_reverse_iterator = std::reverse_iterator<const_iterator>;
214 using size_type = size_t;
215 using difference_type = ptrdiff_t;
216 using absl_internal_is_view = std::true_type;

223 size_type length) noexcept

265 // Implicit constructor from an initializer list, making it possible to pass a
266 // brace-enclosed initializer list to a function expecting a `Span`. Such
267 // spans constructed from an initializer list must be of type `Span<const T>`.
268 //
269 // void Process(absl::Span<const int> x);
270 // Process({1, 2, 3});
271 //
272 // Note that as always the array referenced by the span must outlive the span.
273 // Since an initializer list constructor acts as if it is fed a temporary
274 // array (cf. C++ standard [dcl.init.list]/5), it's safe to use this
275 // constructor only when the `std::initializer_list` itself outlives the span.
276 // In order to meet this requirement it's sufficient to ensure that neither
277 // the span nor a copy of it is used outside of the expression in which it's
278 // created:
279 //
280 // // Assume that this function uses the array directly, not retaining any
281 // // copy of the span or pointer to any of its elements.
282 // void Process(absl::Span<const int> ints);
283 //
284 // // Okay: the std::initializer_list<int> will reference a temporary array
285 // // that isn't destroyed until after the call to Process returns.
286 // Process({ 17, 19 });
287 //
288 // // Not okay: the storage used by the std::initializer_list<int> is not
289 // // allowed to be referenced after the first line.
290 // absl::Span<const int> ints = { 17, 19 };
291 // Process(ints);
292 //
293 // // Not okay for the same reason as above: even when the elements of the
294 // // initializer list expression are not temporaries the underlying array
295 // // is, so the initializer list must still outlive the span.
296 // const int foo = 17;
297 // absl::Span<const int> ints = { foo };
298 // Process(ints);
299 //
300 template <typename LazyT = T,
301 typename = EnableIfValueIsConst<LazyT>>
302 Span(std::initializer_list<value_type> v
303 ABSL_ATTRIBUTE_LIFETIME_BOUND) noexcept // NOLINT(runtime/explicit)
304 : Span(v.begin(), v.size()) {}

329 // Span::operator[]
330 //

332 constexpr reference operator[](size_type i) const noexcept {

335
336 // Span::at()
337 //
338 // Returns a reference to the i'th element of this span.
339 constexpr reference at(size_type i) const {
340 return ABSL_PREDICT_TRUE(i < size()) //
341 ? *(data() + i)
342 : (base_internal::ThrowStdOutOfRange(
343 "Span::at failed bounds check"),
344 *(data() + i));
345 }

347 // Span::front()
348 //
349 // Returns a reference to the first element of this span. The span must not
350 // be empty.
351 constexpr reference front() const noexcept {
352 return ABSL_HARDENING_ASSERT(size() > 0), *data();
353 }
354
355 // Span::back()
356 //
357 // Returns a reference to the last element of this span. The span must not
358 // be empty.
359 constexpr reference back() const noexcept {
360 return ABSL_HARDENING_ASSERT(size() > 0), *(data() + size() - 1);
361 }

419 // Span mutations
420
421 // Span::remove_prefix()
422 //
423 // Removes the first `n` elements from the span.
424 void remove_prefix(size_type n) noexcept {
425 ABSL_HARDENING_ASSERT(size() >= n);
426 ptr_ += n;
427 len_ -= n;
428 }
429
430 // Span::remove_suffix()
431 //
432 // Removes the last `n` elements from the span.
433 void remove_suffix(size_type n) noexcept {
434 ABSL_HARDENING_ASSERT(size() >= n);
435 len_ -= n;
436 }

438 // Span::subspan()
439 //
440 // Returns a `Span` starting at element `pos` and of length `len`. Both `pos`
441 // and `len` are of type `size_type` and thus non-negative. Parameter `pos`
442 // must be <= size(). Any `len` value that points past the end of the span
443 // will be trimmed to at most size() - `pos`. A default `len` value of `npos`
444 // ensures the returned subspan continues until the end of the span.
445 //
446 // Examples:
447 //
448 // std::vector<int> vec = {10, 11, 12, 13};
449 // absl::MakeSpan(vec).subspan(1, 2); // {11, 12}
450 // absl::MakeSpan(vec).subspan(2, 8); // {12, 13}
451 // absl::MakeSpan(vec).subspan(1); // {11, 12, 13}
452 // absl::MakeSpan(vec).subspan(4); // {}
453 // absl::MakeSpan(vec).subspan(5); // throws std::out_of_range
454 constexpr Span subspan(size_type pos = 0, size_type len = npos) const {
455 return (pos <= size())
456 ? Span(data() + pos, (std::min)(size() - pos, len))
457 : (base_internal::ThrowStdOutOfRange("pos > size()"), Span());
458 }
459
460 // Span::first()
461 //
462 // Returns a `Span` containing first `len` elements. Parameter `len` is of
463 // type `size_type` and thus non-negative. `len` value must be <= size().
464 //
465 // Examples:
466 //
467 // std::vector<int> vec = {10, 11, 12, 13};
468 // absl::MakeSpan(vec).first(1); // {10}
469 // absl::MakeSpan(vec).first(3); // {10, 11, 12}
470 // absl::MakeSpan(vec).first(5); // throws std::out_of_range
471 constexpr Span first(size_type len) const {
472 return (len <= size())
473 ? Span(data(), len)
474 : (base_internal::ThrowStdOutOfRange("len > size()"), Span());
475 }
476
477 // Span::last()
478 //
479 // Returns a `Span` containing last `len` elements. Parameter `len` is of
480 // type `size_type` and thus non-negative. `len` value must be <= size().
481 //
482 // Examples:
483 //
484 // std::vector<int> vec = {10, 11, 12, 13};
485 // absl::MakeSpan(vec).last(1); // {13}
486 // absl::MakeSpan(vec).last(3); // {11, 12, 13}
487 // absl::MakeSpan(vec).last(5); // throws std::out_of_range
488 constexpr Span last(size_type len) const {
489 return (len <= size())
490 ? Span(size() - len + data(), len)
491 : (base_internal::ThrowStdOutOfRange("len > size()"), Span());
492 }

494 // Support for absl::Hash.
495 template <typename H>
496 friend H AbslHashValue(H h, Span v) {
497 return H::combine(H::combine_contiguous(std::move(h), v.data(), v.size()),
498 v.size());
499 }

501 private:
502 pointer ptr_;
503 size_type len_;
504};

687// MakeSpan()
688//
689// Constructs a mutable `Span<T>`, deducing `T` automatically from either a
690// container or pointer+size.
691//
692// Because a read-only `Span<const T>` is implicitly constructed from container
693// types regardless of whether the container itself is a const container,
694// constructing mutable spans of type `Span<T>` from containers requires
695// explicit constructors. The container-accepting version of `MakeSpan()`
696// deduces the type of `T` by the constness of the pointer received from the
697// container's `data()` member. Similarly, the pointer-accepting version returns
698// a `Span<const T>` if `T` is `const`, and a `Span<T>` otherwise.
699//
700// Examples:
701//
702// void MyRoutine(absl::Span<MyComplicatedType> a) {
703// ...
704// };
705// // my_vector is a container of non-const types
706// std::vector<MyComplicatedType> my_vector;
707//
708// // Constructing a Span implicitly attempts to create a Span of type
709// // `Span<const T>`
710// MyRoutine(my_vector); // error, type mismatch
711//
712// // Explicitly constructing the Span is verbose
713// MyRoutine(absl::Span<MyComplicatedType>(my_vector));
714//
715// // Use MakeSpan() to make an absl::Span<T>
716// MyRoutine(absl::MakeSpan(my_vector));
717//
718// // Construct a span from an array ptr+size
719// absl::Span<T> my_span() {
720// return absl::MakeSpan(&array[0], num_elements_);
721// }
722//
723template <int&... ExplicitArgumentBarrier, typename T>
724constexpr Span<T> MakeSpan(absl::Nullable<T*> ptr, size_t size) noexcept {
725 return Span<T>(ptr, size);
726}
727
728template <int&... ExplicitArgumentBarrier, typename T>
729Span<T> MakeSpan(absl::Nullable<T*> begin, absl::Nullable<T*> end) noexcept {
730 return ABSL_HARDENING_ASSERT(begin <= end),
731 Span<T>(begin, static_cast<size_t>(end - begin));
732}

735constexpr auto MakeSpan(C& c) noexcept // NOLINT(runtime/references)

737 return MakeSpan(span_internal::GetData(c), c.size());
738}

745// MakeConstSpan()
746//
747// Constructs a `Span<const T>` as with `MakeSpan`, deducing `T` automatically,
748// but always returning a `Span<const T>`.
749//
750// Examples:
751//
752// void ProcessInts(absl::Span<const int> some_ints);
753//
754// // Call with a pointer and size.
755// int array[3] = { 0, 0, 0 };
756// ProcessInts(absl::MakeConstSpan(&array[0], 3));
757//
758// // Call with a [begin, end) pair.
759// ProcessInts(absl::MakeConstSpan(&array[0], &array[3]));
760//
761// // Call directly with an array.
762// ProcessInts(absl::MakeConstSpan(array));
763//
764// // Call with a contiguous container.
765// std::vector<int> some_ints = ...;
766// ProcessInts(absl::MakeConstSpan(some_ints));
767// ProcessInts(absl::MakeConstSpan(std::vector<int>{ 0, 0, 0 }));
768//
769template <int&... ExplicitArgumentBarrier, typename T>
770constexpr Span<const T> MakeConstSpan(absl::Nullable<T*> ptr,
771 size_t size) noexcept {
772 return Span<const T>(ptr, size);
773}
