112 CausalSelfAttention(int block_size, int n_head, int n_embed)
113 : block_size_(block_size), n_head_(n_head), n_embed_(n_embed) {
114 CHECK_EQ(n_embed % n_head, 0);
115
116 // key, query, value projections for all heads, but in a batch
117 c_attn_ = std::make_unique<nn::Linear>(n_embed, 3 * n_embed);
118
119 // output projection
120 c_proj_ = std::make_unique<nn::Linear>(n_embed, n_embed);
121
122 // mask
123 auto dtype = nn::DataTypeToEnum<Type>::value;
124 bias_ = std::make_unique<nn::Parameter>(dtype, block_size * block_size);
125 auto bias_2d = bias_->matrix<Type>(block_size, block_size);
126 nn::UpperTriangularWithNegativeInf(bias_2d);
127
128 // activation tensors
129 qkv_ = std::make_unique<nn::Activation>(dtype); // [B, T, 3C]
130 q_ = std::make_unique<nn::Activation>(dtype); // [B, NH, T, HS]
131 k_ = std::make_unique<nn::Activation>(dtype); // [B, NH, HS, T]
132 v_ = std::make_unique<nn::Activation>(dtype); // [B, NH, T, HS]
133 preatt_ = std::make_unique<nn::Activation>(dtype); // [B, NH, T, T]
134 preatt_softmax_ = std::make_unique<nn::Activation>(dtype); // [B, NH, T, T]
135 att_ = std::make_unique<nn::Activation>(dtype); // [B, NH, T, HS]
136 att2_ = std::make_unique<nn::Activation>(dtype); // [B, T, NH, HS]
137 }

139 void Forward(typename TTypes<Type, 3>::ConstTensor x,
140 typename TTypes<Type, 3>::Tensor y) {
141 PROFILE_TRACE_FN("CausalSelfAttention");
142
143 const int B = x.dimension(0); // batch size
144 const int T = x.dimension(1); // sequence length
145 const int C = x.dimension(2); // embedding dimensionality (n_embd)
146 LOG_FIRST_N(INFO, 1) << "B: " << B << ", T: " << T << ", C: " << C;
147 int NH = n_head_, HS = C / n_head_;
148 CHECK_EQ(B, y.dimension(0));
149 CHECK_EQ(T, y.dimension(1));
150 CHECK(C == n_embed_ && C == y.dimension(2));
151
152 // Lazily allocate the memory for activation
153 qkv_->LazyAllocate(B * T * 3 * C);
154 q_->LazyAllocate(B * NH * T * HS);
155 k_->LazyAllocate(B * NH * HS * T);
156 v_->LazyAllocate(B * NH * T * HS);
157 preatt_->LazyAllocate(B * NH * T * T);
158 preatt_softmax_->LazyAllocate(B * NH * T * T);
159 att_->LazyAllocate(B * NH * T * HS);
160 att2_->LazyAllocate(B * T * NH * HS);
161
162 auto _x = MakeConstMatrix(x.data(), B * T, C);
163 auto qkv = MakeMatrix(qkv_->data<Type>(), B * T, 3 * C);
164 c_attn_->Forward(_x, qkv);
165
166 Eigen::array<Eigen::Index, 3> offsets_q = {0, 0, 0};
167 Eigen::array<Eigen::Index, 3> offsets_k = {0, 0, C};
168 Eigen::array<Eigen::Index, 3> offsets_v = {0, 0, 2 * C};
169 Eigen::array<Eigen::Index, 3> extents = {B, T, C};
170 Eigen::array<Eigen::Index, 4> shape = {B, T, NH, HS};
171 Eigen::array<Eigen::Index, 4> shuffle_qv = {0, 2, 1, 3},
172 shuffle_k = {0, 2, 3, 1};
173 auto qkv3d = qkv_->tensor_3d<Type>(B, T, 3 * C);
174 auto q_4d = q_->tensor_4d<Type>(B, NH, T, HS);
175 auto k_4d = k_->tensor_4d<Type>(B, NH, HS, T);
176 auto v_4d = v_->tensor_4d<Type>(B, NH, T, HS);
177 q_4d.device(nn::g_device) = qkv3d
178 .slice(offsets_q, extents) // [B, T, C]
179 .reshape(shape) // [B, T, NH, HS]
180 .shuffle(shuffle_qv) // [B, NH, T, HS]
181 ;
182 k_4d.device(nn::g_device) = qkv3d
183 .slice(offsets_k, extents) // [B, T, C]
184 .reshape(shape) // [B, T, NH, HS]
185 .shuffle(shuffle_k) // [B, NH, HS, T]
186 ;
187 v_4d.device(nn::g_device) = qkv3d
188 .slice(offsets_v, extents) // [B, T, C]
189 .reshape(shape) // [B, T, NH, HS]
190 .shuffle(shuffle_qv) // [B, NH, T, HS]
191 ;
192
193 const float factor = 1.0f / std::sqrt(static_cast<float>(HS));
194 for (int b = 0; b < B; ++b) {
195 for (int h = 0; h < NH; ++h) {
196 auto q2d =
197 MakeConstMatrix(q_->data<Type>() + (b * NH + h) * T * HS, T, HS);
198 auto k2d =
199 MakeConstMatrix(k_->data<Type>() + (b * NH + h) * HS * T, HS, T);
200 auto v2d = MakeMatrix(v_->data<Type>() + (b * NH + h) * T * HS, T, HS);
201 auto preatt2d =
202 MakeMatrix(preatt_->data<Type>() + (b * NH + h) * T * T, T, T);
203 auto preatt_softmax2d = MakeConstMatrix(
204 preatt_softmax_->data<Type>() + (b * NH + h) * T * T, T, T);
205 auto att2d =
206 MakeMatrix(att_->data<Type>() + (b * NH + h) * T * HS, T, HS);
207
208 nn::MatMul::Forward(q2d, k2d, preatt2d, factor);
209 auto bias_2d = bias_->matrix<Type>(block_size_, block_size_);
210 Eigen::array<Eigen::Index, 2> offsets = {0, 0};
211 Eigen::array<Eigen::Index, 2> extents = {T, T};
212 preatt2d.device(nn::g_device) =
213 preatt2d + bias_2d.slice(offsets, extents);
214
215 // softmax
216 auto preatt2d_tensor =
217 MakeConstMatrix(preatt_->data<Type>() + (b * NH + h) * T * T, T, T);
218 auto preatt_softmax2d_tensor = MakeMatrix(
219 preatt_softmax_->data<Type>() + (b * NH + h) * T * T, T, T);
220 nn::Softmax::Forward(preatt2d_tensor, preatt_softmax2d_tensor);
221
222 // att * v
223 typename TTypes<Type>::ConstMatrix v2d_const =
224 MakeConstMatrix(v_->data<Type>() + (b * NH + h) * T * HS, T, HS);
225 nn::MatMul::Forward(preatt_softmax2d, v2d_const, att2d);
226 }
227 }
228
229 Eigen::array<Eigen::Index, 4> shuffle_att = {0, 2, 1, 3};
230 auto att_4d = att_->tensor_4d<Type>(B, NH, T, HS);
231 auto att2_4d = att2_->tensor_4d<Type>(B, T, NH, HS);
232 att2_4d.device(nn::g_device) =
233 att_4d.shuffle(shuffle_att); // [B, T, NH, HS]
234 auto att2_2d = MakeConstMatrix(att2_->data<Type>(), B * T, C);
235 auto y2d = MakeMatrix(y.data(), B * T, C);
236 c_proj_->Forward(att2_2d, y2d);
237 }

239 void Backward(typename TTypes<Type, 3>::ConstTensor x,
240 typename TTypes<Type, 3>::ConstTensor y_grad,
241 typename TTypes<Type, 3>::Tensor x_grad) {
242 PROFILE_TRACE_FN("CausalSelfAttention");
243
244 const int B = x.dimension(0); // batch size
245 const int T = x.dimension(1); // sequence length
246 const int C = x.dimension(2); // embedding dimensionality (n_embd)
247 int NH = n_head_, HS = C / n_head_;
248 CHECK(B == y_grad.dimension(0) && B == x_grad.dimension(0));
249 CHECK(T == y_grad.dimension(1) && T == x_grad.dimension(1));
250 CHECK(C == y_grad.dimension(2) && C == x_grad.dimension(2));
251
252 // Lazily allocate the memory for activation
253 qkv_->LazyAllocateGradient();
254 q_->LazyAllocateGradient();
255 k_->LazyAllocateGradient();
256 v_->LazyAllocateGradient();
257 preatt_->LazyAllocateGradient();
258 preatt_softmax_->LazyAllocateGradient();
259 att_->LazyAllocateGradient();
260 att2_->LazyAllocateGradient();
261 qkv_->ZeroGrad();
262 q_->ZeroGrad();
263 k_->ZeroGrad();
264 v_->ZeroGrad();
265 preatt_->ZeroGrad();
266 preatt_softmax_->ZeroGrad();
267 att_->ZeroGrad();
268 att2_->ZeroGrad();
269
270 // attproj backward
271 auto att2_2d = MakeConstMatrix(att2_->data<Type>(), B * T, C);
272 auto y_grad_2d = MakeConstMatrix(y_grad.data(), B * T, C);
273 auto att2_grad_2d = MakeMatrix(att2_->grad<Type>(), B * T, C);
274 c_proj_->Backward(att2_2d, y_grad_2d, att2_grad_2d);
275
276 // shuffle backward
277 Eigen::array<Eigen::Index, 4> shuffle_att = {0, 2, 1, 3};
278 auto att_grad = att_->tensor_4d_grad<Type>(B, NH, T, HS);
279 auto att2_grad = att2_->tensor_4d_grad<Type>(B, T, NH, HS);
280 att_grad.device(nn::g_device) =
281 att2_grad.shuffle(shuffle_att); // [B, NH, T, HS]
282
283 // attention backward
284 float factor = 1.0f / std::sqrt(static_cast<float>(HS));
285 for (int b = 0; b < B; ++b) {
286 for (int h = 0; h < NH; ++h) {
287 auto q2d =
288 MakeConstMatrix(q_->data<Type>() + (b * NH + h) * T * HS, T, HS);
289 auto q_grad2d =
290 MakeMatrix(q_->grad<Type>() + (b * NH + h) * T * HS, T, HS);
291 auto k2d =
292 MakeConstMatrix(k_->data<Type>() + (b * NH + h) * HS * T, HS, T);
293 auto k_grad2d =
294 MakeMatrix(k_->grad<Type>() + (b * NH + h) * HS * T, HS, T);
295 auto v2d =
296 MakeConstMatrix(v_->data<Type>() + (b * NH + h) * T * HS, T, HS);
297 auto v_grad2d =
298 MakeMatrix(v_->grad<Type>() + (b * NH + h) * T * HS, T, HS);
299 auto preatt2d =
300 MakeConstMatrix(preatt_->data<Type>() + (b * NH + h) * T * T, T, T);
301 auto preatt_softmax2d = MakeConstMatrix(
302 preatt_softmax_->data<Type>() + (b * NH + h) * T * T, T, T);
303 auto preatt_grad2d =
304 MakeMatrix(preatt_->grad<Type>() + (b * NH + h) * T * T, T, T);
305 auto preatt_grad2d_const =
306 MakeConstMatrix(preatt_->grad<Type>() + (b * NH + h) * T * T, T, T);
307 auto preatt_softmax_grad2d = MakeMatrix(
308 preatt_softmax_->grad<Type>() + (b * NH + h) * T * T, T, T);
309 auto preatt_softmax_grad2d_const = MakeConstMatrix(
310 preatt_softmax_->grad<Type>() + (b * NH + h) * T * T, T, T);
311 auto att_grad2d =
312 MakeConstMatrix(att_->grad<Type>() + (b * NH + h) * T * HS, T, HS);
313
314 // backward: att * v
315 nn::MatMul::Backward(preatt_softmax2d, v2d, att_grad2d,
316 preatt_softmax_grad2d, v_grad2d);
317
318 // backward: softmax
319 nn::Softmax::Backward(preatt_softmax2d, preatt_softmax_grad2d_const,
320 preatt_grad2d);
321
322 // backward: mask
323 // backward: q * k
324 nn::MatMul::Backward(q2d, k2d, preatt_grad2d_const, q_grad2d, k_grad2d,
325 factor);
326 }
327 }
328
329 // backward: shuffle -> reshape
330 Eigen::array<Eigen::Index, 3> offsets_q = {0, 0, 0};
331 Eigen::array<Eigen::Index, 3> offsets_k = {0, 0, C};
332 Eigen::array<Eigen::Index, 3> offsets_v = {0, 0, 2 * C};
333 Eigen::array<Eigen::Index, 3> extents = {B, T, C};
334 Eigen::array<Eigen::Index, 3> shape = {B, T, C};
335 Eigen::array<Eigen::Index, 4> shuffle_qv = {0, 2, 1, 3},
336 shuffle_k = {0, 3, 1, 2};
337 // q_grad_: [B, NH, T, HS] -> [B, T, NH, HS] -> [B, T, C]
338 auto qkv_grad = qkv_->tensor_3d_grad<Type>(B, T, 3 * C);
339 auto q_grad = q_->tensor_4d_grad<Type>(B, NH, T, HS);
340 auto k_grad = k_->tensor_4d_grad<Type>(B, NH, HS, T);
341 auto v_grad = v_->tensor_4d_grad<Type>(B, NH, T, HS);
342 qkv_grad.slice(offsets_q, extents).device(nn::g_device) =
343 q_grad.shuffle(shuffle_qv).reshape(shape);
344
345 // k_grad_: [B, NH, HS, T] -> [B, T, NH, HS] -> [B, T, C]
346 qkv_grad.slice(offsets_k, extents).device(nn::g_device) =
347 k_grad.shuffle(shuffle_k).reshape(shape);
348
349 // v_grad_: [B, NH, T, HS] -> [B, T, NH, HS] -> [B, T, C]
350 qkv_grad.slice(offsets_v, extents).device(nn::g_device) =
351 v_grad.shuffle(shuffle_qv).reshape(shape);
352
353 // backward: qkv
354 auto _x = MakeConstMatrix(x.data(), B * T, C);
355 auto qkv_grad_2d = MakeConstMatrix(qkv_->grad<Type>(), B * T, 3 * C);
356 auto _x_grad = MakeMatrix(x_grad.data(), B * T, C);
357 c_attn_->Backward(_x, qkv_grad_2d, _x_grad);
358 }

467 void Forward(typename TTypes<Type, 3>::ConstTensor x,
468 typename TTypes<Type, 3>::Tensor y) {
469 PROFILE_TRACE_FN("Block");
470
471 // x: [B, T, C], y: [B, T, C]
472 const int B = x.dimension(0); // batch size
473 const int T = x.dimension(1); // sequence length
474 const int C = x.dimension(2); // embedding dimensionality (n_embd)
475 CHECK_EQ(B, y.dimension(0));
476 CHECK_EQ(T, y.dimension(1));
477 CHECK_EQ(C, y.dimension(2));
478
479 ln1_y_->LazyAllocate(B * T * C);
480 ln1_mean_->LazyAllocate(B * T);
481 ln1_rstd_->LazyAllocate(B * T);
482 att_y_->LazyAllocate(B * T * C);
483 residual1_->LazyAllocate(B * T * C);
484 ln2_y_->LazyAllocate(B * T * C);
485 ln2_mean_->LazyAllocate(B * T);
486 ln2_rstd_->LazyAllocate(B * T);
487 mlp_y_->LazyAllocate(B * T * C);
488
489 // LN1
490 auto x_2d = MakeConstMatrix(x.data(), B * T, C);
491 auto ln1_y_2d = MakeMatrix(ln1_y_->data<Type>(), B * T, C);
492 auto ln1_mean_1d = MakeFlat(ln1_mean_->data<Type>(), B * T);
493 auto ln1_rstd_1d = MakeFlat(ln1_rstd_->data<Type>(), B * T);
494 ln1_->Forward(x_2d, ln1_y_2d, ln1_mean_1d, ln1_rstd_1d);
495
496 // Attention
497 auto ln1_y_3d = MakeConst3DTensor(ln1_y_2d.data(), B, T, C);
498 auto att_y_3d = Make3DTensor(att_y_->data<Type>(), B, T, C);
499 attn_->Forward(ln1_y_3d, att_y_3d);
500
501 // Residual
502 auto x_1d = MakeConstFlat(x.data(), B * T * C);
503 auto att_y_1d = MakeConstFlat(att_y_->data<Type>(), B * T * C);
504 auto residual1_1d = MakeFlat(residual1_->data<Type>(), residual1_->size());
505 nn::Residual::Forward(x_1d, att_y_1d, residual1_1d);
506
507 // LN2
508 auto ln2_y_2d = MakeMatrix(ln2_y_->data<Type>(), B * T, C);
509 auto ln2_y_2d_const = MakeConstMatrix(ln2_y_->data<Type>(), B * T, C);
510 auto ln2_mean_1d = MakeFlat(ln2_mean_->data<Type>(), B * T);
511 auto ln2_rstd_1d = MakeFlat(ln2_rstd_->data<Type>(), B * T);
512 auto residual1_2d = MakeConstMatrix(residual1_->data<Type>(), B * T, C);
513 ln2_->Forward(residual1_2d, ln2_y_2d, ln2_mean_1d, ln2_rstd_1d);
514
515 // MLP
516 auto mlp_y_2d = MakeMatrix(mlp_y_->data<Type>(), B * T, C);
517 mlp_->Forward(ln2_y_2d_const, mlp_y_2d);
518
519 // Residual
520 auto residual1_1d_const =
521 MakeConstFlat(residual1_->data<Type>(), residual1_->size());
522 auto mlp_y_1d = MakeConstFlat(mlp_y_->data<Type>(), B * T * C);
523 auto y_1d = MakeFlat(y.data(), y.size());
524 nn::Residual::Forward(residual1_1d_const, mlp_y_1d, y_1d);
525 }

527 void Backward(typename TTypes<Type, 3>::ConstTensor x,
528 typename TTypes<Type, 3>::ConstTensor y_grad,
529 typename TTypes<Type, 3>::Tensor x_grad) {
530 PROFILE_TRACE_FN("Block");
531
532 // x: [B, T, C], y_grad: [B, T, C], x_grad: [B, T, C]
533 const int B = x.dimension(0); // batch size
534 const int T = x.dimension(1); // sequence length
535 const int C = x.dimension(2); // embedding dimensionality (n_embd)
536 CHECK_EQ(B, y_grad.dimension(0));
537 CHECK_EQ(T, y_grad.dimension(1));
538 CHECK_EQ(C, y_grad.dimension(2));
539 CHECK_EQ(B, x_grad.dimension(0));
540 CHECK_EQ(T, x_grad.dimension(1));
541 CHECK_EQ(C, x_grad.dimension(2));
542
543 ln1_y_->LazyAllocateGradient();
544 att_y_->LazyAllocateGradient();
545 residual1_->LazyAllocateGradient();
546 ln2_y_->LazyAllocateGradient();
547 mlp_y_->LazyAllocateGradient();
548 ln1_y_->ZeroGrad();
549 att_y_->ZeroGrad();
550 residual1_->ZeroGrad();
551 ln2_y_->ZeroGrad();
552 mlp_y_->ZeroGrad();
553
554 // backward residual
555 auto y_grad_1d = MakeConstFlat(y_grad.data(), y_grad.size());
556 auto residual1_grad_1d =
557 MakeFlat(residual1_->grad<Type>(), residual1_->size());
558 auto mlp_y_grad_1d = MakeFlat(mlp_y_->grad<Type>(), mlp_y_->size());
559 nn::Residual::Backward(y_grad_1d, residual1_grad_1d, mlp_y_grad_1d);
560
561 // backward MLP
562 auto ln2_y_2d = MakeConstMatrix(ln2_y_->data<Type>(), B * T, C);
563 auto ln2_y_grad_2d = MakeMatrix(ln2_y_->grad<Type>(), B * T, C);
564 auto mlp_y_grad_2d = MakeConstMatrix(mlp_y_->grad<Type>(), B * T, C);
565 mlp_->Backward(ln2_y_2d, mlp_y_grad_2d, ln2_y_grad_2d);
566
567 // backward LN2
568 auto ln2_mean_1d = MakeConstFlat(ln2_mean_->data<Type>(), B * T);
569 auto ln2_rstd_1d = MakeConstFlat(ln2_rstd_->data<Type>(), B * T);
570 auto residual1_2d = MakeConstMatrix(residual1_->data<Type>(), B * T, C);
571 auto ln2_y_grad_2d_const = MakeConstMatrix(ln2_y_->grad<Type>(), B * T, C);
572 auto residual1_grad_2d = MakeMatrix(residual1_->grad<Type>(), B * T, C);
573 ln2_->Backward(residual1_2d, ln2_y_grad_2d_const, ln2_mean_1d, ln2_rstd_1d,
574 residual1_grad_2d);
575
576 // backward residual
577 auto residual1_grad_1d_const =
578 MakeConstFlat(residual1_->grad<Type>(), residual1_->size());
579 auto x_grad_1d = MakeFlat(x_grad.data(), x_grad.size());
580 auto att_y_grad_1d = MakeFlat(att_y_->grad<Type>(), att_y_->size());
581 nn::Residual::Backward(residual1_grad_1d_const, x_grad_1d, att_y_grad_1d);
582
583 // backward attention
584 auto ln1_y_3d = MakeConst3DTensor(ln1_y_->data<Type>(), B, T, C);
585 auto ln1_y_grad_3d = Make3DTensor(ln1_y_->grad<Type>(), B, T, C);
586 auto att_y_grad_3d = MakeConst3DTensor(att_y_->grad<Type>(), B, T, C);
587 attn_->Backward(ln1_y_3d, att_y_grad_3d, ln1_y_grad_3d);
588
589 // backward LN1
590 auto x_2d = MakeConstMatrix(x.data(), B * T, C);
591 auto ln1_mean_1d = MakeConstFlat(ln1_mean_->data<Type>(), B * T);
592 auto ln1_rstd_1d = MakeConstFlat(ln1_rstd_->data<Type>(), B * T);
593 auto ln1_y_grad_2d = MakeConstMatrix(ln1_y_->grad<Type>(), B * T, C);
594 auto x_grad_2d = MakeMatrix(x_grad.data(), B * T, C);
595 ln1_->Backward(x_2d, ln1_y_grad_2d, ln1_mean_1d, ln1_rstd_1d, x_grad_2d);
596 }

599 return ln1_->NumParameters() + attn_->NumParameters() +

601 }

603 size_t NumActivations() const {
604 return ln1_->NumActivations() + attn_->NumActivations() +

606 ln1_mean_->size() + ln2_rstd_->size() + att_y_->size() +
607 residual1_->size() + ln2_y_->size() + ln2_mean_->size() +
608 ln2_rstd_->size() + mlp_y_->size();
609 }

619 std::unique_ptr<CausalSelfAttention> attn_;

631};

682 void __Forward(typename TTypes<int>::ConstMatrix idx) {
683 PROFILE_TRACE_FN("GPT");
684
685 const int B = idx.dimension(0), T = idx.dimension(1), C = n_embed_,
686 L = n_layer_;
687 const int BT = B * T, TC = T * C;
688 const int BTC = BT * C;
689
690 CHECK_LE(T, block_size_) << "Cannot forward sequence of length " << T
691 << ", block size is only " << block_size_;
692 std::vector<int> pos(T);
693 std::iota(pos.begin(), pos.end(), 0);
694
695 // Lazily allocate memory
696 tok_emb_->LazyAllocate(B * T * C);
697 pos_emb_->LazyAllocate(T * C);
698 encoded_->LazyAllocate(B * T * C);
699 block_y_->LazyAllocate(L * B * T * C);
700 lnf_y_->LazyAllocate(BT * C);
701 lnf_mean_->LazyAllocate(BT);
702 lnf_rstd_->LazyAllocate(BT);
703
704 wte_->Forward(idx,
705 absl::MakeSpan(tok_emb_->data<Type>(), tok_emb_->size()));
706 wpe_->Forward(pos,
707 absl::MakeSpan(pos_emb_->data<Type>(), pos_emb_->size()));
708
709 auto tok_emb = tok_emb_->matrix<Type>(B, TC);
710 auto pos_emb = pos_emb_->flat<Type>();
711 auto encoded = encoded_->matrix<Type>(B, TC);
712 Eigen::array<Eigen::Index, 2> batch_by_one = {B, 1};
713 Eigen::array<Eigen::Index, 2> one_by_class = {1, TC};
714 encoded.device(nn::g_device) =
715 tok_emb + pos_emb.reshape(one_by_class).broadcast(batch_by_one);
716
717 for (int l = 0; l < n_layer_; ++l) {
718 const auto& block = h_[l];
719 Type* x = l == 0 ? encoded_->data<Type>()
720 : block_y_->data<Type>() + (l - 1) * BTC;
721 Type* y = block_y_->data<Type>() + l * BTC;
722 auto block_x_3d = MakeConst3DTensor(x, B, T, C);
723 auto block_y_3d = Make3DTensor(y, B, T, C);
724 block->Forward(block_x_3d, block_y_3d);
725 }
726
727 auto block_out_2d =
728 MakeConstMatrix(block_y_->data<Type>() + (L - 1) * BTC, BT, C);
729 auto lnf_y = MakeMatrix(lnf_y_->data<Type>(), BT, C);
730 auto lnf_mean = MakeFlat(lnf_mean_->data<Type>(), BT);
731 auto lnf_rstd = MakeFlat(lnf_rstd_->data<Type>(), BT);
732 lnf_->Forward(block_out_2d, lnf_y, lnf_mean, lnf_rstd);
733 }

898 tok_emb_->LazyAllocateGradient();
899 pos_emb_->LazyAllocateGradient();
900 encoded_->LazyAllocateGradient();
901 block_y_->LazyAllocateGradient();
902 lnf_y_->LazyAllocateGradient();
903
904 tok_emb_->ZeroGrad();
905 pos_emb_->ZeroGrad();
906 encoded_->ZeroGrad();
907 block_y_->ZeroGrad();
908 lnf_y_->ZeroGrad();
909
910 // backward lm_head
911 auto logits_grad_2d =
912 MakeMatrix(logits_grad_->data<Type>(), BT, vocab_size_);

914 auto lnf_y_grad = MakeMatrix(lnf_y_->grad<Type>(), BT, C);
915 auto lm_head = MakeMatrix(lm_head_, vocab_size_, C);
916 auto lm_head_grad = MakeMatrix(lm_head_grad_, vocab_size_, C);
917 Eigen::array<Eigen::IndexPair<int>, 1> product_dims = {
918 Eigen::IndexPair<int>(1, 0)};
919 Eigen::array<Eigen::IndexPair<int>, 1> product_dims2 = {
920 Eigen::IndexPair<int>(0, 0)};
921 lnf_y_grad.device(nn::g_device) += logits_grad_2d.contract(
922 lm_head, product_dims); // [BT, vocab_size] x [vocab_size, C]
923 lm_head_grad.device(nn::g_device) += logits_grad_2d.contract(
924 lnf_y, product_dims2); // [vocab_size, BT] x [BT, C]
925
926 // backward LNF
927 auto block_out_2d =
928 MakeConstMatrix(block_y_->data<Type>() + (L - 1) * BTC, BT, C);
929 auto block_out_grad_2d =
930 MakeMatrix(block_y_->grad<Type>() + (L - 1) * BTC, BT, C);
931 auto lnf_mean = MakeConstFlat(lnf_mean_->data<Type>(), BT);
932 auto lnf_rstd = MakeConstFlat(lnf_rstd_->data<Type>(), BT);
933 auto lnf_y_grad_2d = MakeConstMatrix(lnf_y_->grad<Type>(), BT, C);
934 lnf_->Backward(block_out_2d, lnf_y_grad_2d, lnf_mean, lnf_rstd,

962 wte_->Backward(idx, tok_emb_grad);

964 }
