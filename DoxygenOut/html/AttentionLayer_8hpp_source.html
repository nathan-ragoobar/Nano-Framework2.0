26 CausalSelfAttention(int block_size, int n_head, int n_embed)
27 : block_size_(block_size), n_head_(n_head), n_embed_(n_embed) {
28 CHECK_EQ(n_embed % n_head, 0);
29
30 // key, query, value projections for all heads, but in a batch
31 c_attn_ = std::make_unique<nn::Linear>(n_embed, 3 * n_embed);
32
33 // output projection
34 c_proj_ = std::make_unique<nn::Linear>(n_embed, n_embed);
35
36 // mask
37 auto dtype = nn::DataTypeToEnum<Type>::value;
38 bias_ = std::make_unique<nn::Parameter>(dtype, block_size * block_size);
39 auto bias_2d = bias_->matrix<Type>(block_size, block_size);
40 nn::UpperTriangularWithNegativeInf(bias_2d);
41
42 // activation tensors
43 qkv_ = std::make_unique<nn::Activation>(dtype); // [B, T, 3C]
44 q_ = std::make_unique<nn::Activation>(dtype); // [B, NH, T, HS]
45 k_ = std::make_unique<nn::Activation>(dtype); // [B, NH, HS, T]
46 v_ = std::make_unique<nn::Activation>(dtype); // [B, NH, T, HS]
47 preatt_ = std::make_unique<nn::Activation>(dtype); // [B, NH, T, T]
48 preatt_softmax_ = std::make_unique<nn::Activation>(dtype); // [B, NH, T, T]
49 att_ = std::make_unique<nn::Activation>(dtype); // [B, NH, T, HS]
50 att2_ = std::make_unique<nn::Activation>(dtype); // [B, T, NH, HS]
51 }

53 void Forward(typename TTypes<Type, 3>::ConstTensor x,
54 typename TTypes<Type, 3>::Tensor y) {
55 PROFILE_TRACE_FN("CausalSelfAttention");
56
57 const int B = x.dimension(0); // batch size
58 const int T = x.dimension(1); // sequence length
59 const int C = x.dimension(2); // embedding dimensionality (n_embd)
60 LOG_FIRST_N(INFO, 1) << "B: " << B << ", T: " << T << ", C: " << C;
61 int NH = n_head_, HS = C / n_head_;
62 CHECK_EQ(B, y.dimension(0));
63 CHECK_EQ(T, y.dimension(1));
64 CHECK(C == n_embed_ && C == y.dimension(2));
65
66 // Lazily allocate the memory for activation
67 qkv_->LazyAllocate(B * T * 3 * C);
68 q_->LazyAllocate(B * NH * T * HS);
69 k_->LazyAllocate(B * NH * HS * T);
70 v_->LazyAllocate(B * NH * T * HS);
71 preatt_->LazyAllocate(B * NH * T * T);
72 preatt_softmax_->LazyAllocate(B * NH * T * T);
73 att_->LazyAllocate(B * NH * T * HS);
74 att2_->LazyAllocate(B * T * NH * HS);
75
76 auto _x = MakeConstMatrix(x.data(), B * T, C);
77 auto qkv = MakeMatrix(qkv_->data<Type>(), B * T, 3 * C);
78 c_attn_->Forward(_x, qkv);
79
80 Eigen::array<Eigen::Index, 3> offsets_q = {0, 0, 0};
81 Eigen::array<Eigen::Index, 3> offsets_k = {0, 0, C};
82 Eigen::array<Eigen::Index, 3> offsets_v = {0, 0, 2 * C};
83 Eigen::array<Eigen::Index, 3> extents = {B, T, C};
84 Eigen::array<Eigen::Index, 4> shape = {B, T, NH, HS};
85 Eigen::array<Eigen::Index, 4> shuffle_qv = {0, 2, 1, 3},
86 shuffle_k = {0, 2, 3, 1};
87 auto qkv3d = qkv_->tensor_3d<Type>(B, T, 3 * C);
88 auto q_4d = q_->tensor_4d<Type>(B, NH, T, HS);
89 auto k_4d = k_->tensor_4d<Type>(B, NH, HS, T);
90 auto v_4d = v_->tensor_4d<Type>(B, NH, T, HS);
91 q_4d.device(nn::g_device) = qkv3d
92 .slice(offsets_q, extents) // [B, T, C]
93 .reshape(shape) // [B, T, NH, HS]
94 .shuffle(shuffle_qv) // [B, NH, T, HS]
95 ;
96 k_4d.device(nn::g_device) = qkv3d
97 .slice(offsets_k, extents) // [B, T, C]
98 .reshape(shape) // [B, T, NH, HS]
99 .shuffle(shuffle_k) // [B, NH, HS, T]
100 ;
101 v_4d.device(nn::g_device) = qkv3d
102 .slice(offsets_v, extents) // [B, T, C]
103 .reshape(shape) // [B, T, NH, HS]
104 .shuffle(shuffle_qv) // [B, NH, T, HS]
105 ;
106
107 const float factor = 1.0f / std::sqrt(static_cast<float>(HS));
108 for (int b = 0; b < B; ++b) {
109 for (int h = 0; h < NH; ++h) {
110 auto q2d =
111 MakeConstMatrix(q_->data<Type>() + (b * NH + h) * T * HS, T, HS);
112 auto k2d =
113 MakeConstMatrix(k_->data<Type>() + (b * NH + h) * HS * T, HS, T);
114 auto v2d = MakeMatrix(v_->data<Type>() + (b * NH + h) * T * HS, T, HS);
115 auto preatt2d =
116 MakeMatrix(preatt_->data<Type>() + (b * NH + h) * T * T, T, T);
117 auto preatt_softmax2d = MakeConstMatrix(
118 preatt_softmax_->data<Type>() + (b * NH + h) * T * T, T, T);
119 auto att2d =
120 MakeMatrix(att_->data<Type>() + (b * NH + h) * T * HS, T, HS);
121
122 nn::MatMul::Forward(q2d, k2d, preatt2d, factor);
123 auto bias_2d = bias_->matrix<Type>(block_size_, block_size_);
124 Eigen::array<Eigen::Index, 2> offsets = {0, 0};
125 Eigen::array<Eigen::Index, 2> extents = {T, T};
126 preatt2d.device(nn::g_device) =
127 preatt2d + bias_2d.slice(offsets, extents);
128
129 // softmax
130 auto preatt2d_tensor =
131 MakeConstMatrix(preatt_->data<Type>() + (b * NH + h) * T * T, T, T);
132 auto preatt_softmax2d_tensor = MakeMatrix(
133 preatt_softmax_->data<Type>() + (b * NH + h) * T * T, T, T);
134 nn::Softmax::Forward(preatt2d_tensor, preatt_softmax2d_tensor);
135
136 // att * v
137 typename TTypes<Type>::ConstMatrix v2d_const =
138 MakeConstMatrix(v_->data<Type>() + (b * NH + h) * T * HS, T, HS);
139 nn::MatMul::Forward(preatt_softmax2d, v2d_const, att2d);
140 }
141 }
142
143 Eigen::array<Eigen::Index, 4> shuffle_att = {0, 2, 1, 3};
144 auto att_4d = att_->tensor_4d<Type>(B, NH, T, HS);
145 auto att2_4d = att2_->tensor_4d<Type>(B, T, NH, HS);
146 att2_4d.device(nn::g_device) =
147 att_4d.shuffle(shuffle_att); // [B, T, NH, HS]
148 auto att2_2d = MakeConstMatrix(att2_->data<Type>(), B * T, C);
149 auto y2d = MakeMatrix(y.data(), B * T, C);
150 c_proj_->Forward(att2_2d, y2d);
151 }

153 void Backward(typename TTypes<Type, 3>::ConstTensor x,
154 typename TTypes<Type, 3>::ConstTensor y_grad,
155 typename TTypes<Type, 3>::Tensor x_grad) {
156 PROFILE_TRACE_FN("CausalSelfAttention");
157
158 const int B = x.dimension(0); // batch size
159 const int T = x.dimension(1); // sequence length
160 const int C = x.dimension(2); // embedding dimensionality (n_embd)
161 int NH = n_head_, HS = C / n_head_;
162 CHECK(B == y_grad.dimension(0) && B == x_grad.dimension(0));
163 CHECK(T == y_grad.dimension(1) && T == x_grad.dimension(1));
164 CHECK(C == y_grad.dimension(2) && C == x_grad.dimension(2));
165
166 // Lazily allocate the memory for activation
167 qkv_->LazyAllocateGradient();
168 q_->LazyAllocateGradient();
169 k_->LazyAllocateGradient();
170 v_->LazyAllocateGradient();
171 preatt_->LazyAllocateGradient();
172 preatt_softmax_->LazyAllocateGradient();
173 att_->LazyAllocateGradient();
174 att2_->LazyAllocateGradient();
175 qkv_->ZeroGrad();
176 q_->ZeroGrad();
177 k_->ZeroGrad();
178 v_->ZeroGrad();
179 preatt_->ZeroGrad();
180 preatt_softmax_->ZeroGrad();
181 att_->ZeroGrad();
182 att2_->ZeroGrad();
183
184 // attproj backward
185 auto att2_2d = MakeConstMatrix(att2_->data<Type>(), B * T, C);
186 auto y_grad_2d = MakeConstMatrix(y_grad.data(), B * T, C);
187 auto att2_grad_2d = MakeMatrix(att2_->grad<Type>(), B * T, C);
188 c_proj_->Backward(att2_2d, y_grad_2d, att2_grad_2d);
189
190 // shuffle backward
191 Eigen::array<Eigen::Index, 4> shuffle_att = {0, 2, 1, 3};
192 auto att_grad = att_->tensor_4d_grad<Type>(B, NH, T, HS);
193 auto att2_grad = att2_->tensor_4d_grad<Type>(B, T, NH, HS);
194 att_grad.device(nn::g_device) =
195 att2_grad.shuffle(shuffle_att); // [B, NH, T, HS]
196
197 // attention backward
198 float factor = 1.0f / std::sqrt(static_cast<float>(HS));
199 for (int b = 0; b < B; ++b) {
200 for (int h = 0; h < NH; ++h) {
201 auto q2d =
202 MakeConstMatrix(q_->data<Type>() + (b * NH + h) * T * HS, T, HS);
203 auto q_grad2d =
204 MakeMatrix(q_->grad<Type>() + (b * NH + h) * T * HS, T, HS);
205 auto k2d =
206 MakeConstMatrix(k_->data<Type>() + (b * NH + h) * HS * T, HS, T);
207 auto k_grad2d =
208 MakeMatrix(k_->grad<Type>() + (b * NH + h) * HS * T, HS, T);
209 auto v2d =
210 MakeConstMatrix(v_->data<Type>() + (b * NH + h) * T * HS, T, HS);
211 auto v_grad2d =
212 MakeMatrix(v_->grad<Type>() + (b * NH + h) * T * HS, T, HS);
213 auto preatt2d =
214 MakeConstMatrix(preatt_->data<Type>() + (b * NH + h) * T * T, T, T);
215 auto preatt_softmax2d = MakeConstMatrix(
216 preatt_softmax_->data<Type>() + (b * NH + h) * T * T, T, T);
217 auto preatt_grad2d =
218 MakeMatrix(preatt_->grad<Type>() + (b * NH + h) * T * T, T, T);
219 auto preatt_grad2d_const =
220 MakeConstMatrix(preatt_->grad<Type>() + (b * NH + h) * T * T, T, T);
221 auto preatt_softmax_grad2d = MakeMatrix(
222 preatt_softmax_->grad<Type>() + (b * NH + h) * T * T, T, T);
223 auto preatt_softmax_grad2d_const = MakeConstMatrix(
224 preatt_softmax_->grad<Type>() + (b * NH + h) * T * T, T, T);
225 auto att_grad2d =
226 MakeConstMatrix(att_->grad<Type>() + (b * NH + h) * T * HS, T, HS);
227
228 // backward: att * v
229 nn::MatMul::Backward(preatt_softmax2d, v2d, att_grad2d,
230 preatt_softmax_grad2d, v_grad2d);
231
232 // backward: softmax
233 nn::Softmax::Backward(preatt_softmax2d, preatt_softmax_grad2d_const,
234 preatt_grad2d);
235
236 // backward: mask
237 // backward: q * k
238 nn::MatMul::Backward(q2d, k2d, preatt_grad2d_const, q_grad2d, k_grad2d,
239 factor);
240 }
241 }
242
243 // backward: shuffle -> reshape
244 Eigen::array<Eigen::Index, 3> offsets_q = {0, 0, 0};
245 Eigen::array<Eigen::Index, 3> offsets_k = {0, 0, C};
246 Eigen::array<Eigen::Index, 3> offsets_v = {0, 0, 2 * C};
247 Eigen::array<Eigen::Index, 3> extents = {B, T, C};
248 Eigen::array<Eigen::Index, 3> shape = {B, T, C};
249 Eigen::array<Eigen::Index, 4> shuffle_qv = {0, 2, 1, 3},
250 shuffle_k = {0, 3, 1, 2};
251 // q_grad_: [B, NH, T, HS] -> [B, T, NH, HS] -> [B, T, C]
252 auto qkv_grad = qkv_->tensor_3d_grad<Type>(B, T, 3 * C);
253 auto q_grad = q_->tensor_4d_grad<Type>(B, NH, T, HS);
254 auto k_grad = k_->tensor_4d_grad<Type>(B, NH, HS, T);
255 auto v_grad = v_->tensor_4d_grad<Type>(B, NH, T, HS);
256 qkv_grad.slice(offsets_q, extents).device(nn::g_device) =
257 q_grad.shuffle(shuffle_qv).reshape(shape);
258
259 // k_grad_: [B, NH, HS, T] -> [B, T, NH, HS] -> [B, T, C]
260 qkv_grad.slice(offsets_k, extents).device(nn::g_device) =
261 k_grad.shuffle(shuffle_k).reshape(shape);
262
263 // v_grad_: [B, NH, T, HS] -> [B, T, NH, HS] -> [B, T, C]
264 qkv_grad.slice(offsets_v, extents).device(nn::g_device) =
265 v_grad.shuffle(shuffle_qv).reshape(shape);
266
267 // backward: qkv
268 auto _x = MakeConstMatrix(x.data(), B * T, C);
269 auto qkv_grad_2d = MakeConstMatrix(qkv_->grad<Type>(), B * T, 3 * C);
270 auto _x_grad = MakeMatrix(x_grad.data(), B * T, C);
271 c_attn_->Backward(_x, qkv_grad_2d, _x_grad);
272 }
