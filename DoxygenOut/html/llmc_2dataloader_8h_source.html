61int64_t dataloader_load_shard_(DataLoader *loader, int shard_index) {
62 if (loader->should_shuffle) {
63 shard_index = loader->shard_indices[shard_index];
64 }
65 // use the first glob match as the filename for now
66 const char* filename = loader->glob_result.gl_pathv[shard_index];
67 // open the input file for reading. also only a single file can be opened at a time
68 if (loader->tokens_file != NULL) {
69 fcloseCheck(loader->tokens_file);
70 }
71 loader->tokens_file = fopenCheck(filename, "rb");
72 // validate the header
73 int header[HEADER_SIZE];
74 freadCheck(header, sizeof(int), HEADER_SIZE, loader->tokens_file);
75 if (header[0] != 20240520) {
76 printf("Bad magic in the data file\n");
77 printf("---> HINT: Are you passing in a correct file?\n");
78 printf("---> HINT: The data encoding may have changed, re-run data prepro or refer again to README.\n");
79 exit(EXIT_FAILURE);
80 }
81 if (header[1] != 1) { printf("Bad version in data file\n"); exit(EXIT_FAILURE); }
82 int64_t ntok = header[2]; // number of tokens in the file
83 assert(ntok > 0); // we expect some tokens in the file. this should never trip, right?
84 // determine the file size and make sure it is consistent with the number of tokens
85 fseekCheck(loader->tokens_file, 0, SEEK_END); // seek to end of file
86 loader->file_size_bytes = ftell(loader->tokens_file); // read the offset, i.e. file size
87 fseekCheck(loader->tokens_file, 0, SEEK_SET); // seek back to the beginning
88 // we expect ntok in the file to be consistent with filesize, assert that is the case
89 int64_t expected_file_size = HEADER_SIZE * sizeof(int) + ntok * sizeof(uint16_t);
90 if (loader->file_size_bytes != expected_file_size) {
91 printf("Error: file size is not as expected\n");
92 exit(EXIT_FAILURE);
93 }
94 // -1 uint16_t due to us taking B*T+1 tokens but moving by B*T tokens
95 loader->shard_num_samples = (ntok * sizeof(uint16_t) - sizeof(uint16_t)) / loader->total_batch_size_bytes;
96 return ntok;
97}

111 loader->current_shard_idx = 0;

113

125void dataloader_advance_(DataLoader *loader) {
126 if (loader->current_shard_idx == loader->glob_result.gl_pathc - 1) {
127 // if we are at the last shard, we reset the loader and start a new epoch
128 dataloader_reset(loader);
129 return;
130 }
131
132 // advance the loader by loading the next data shard and resetting the position
133 loader->current_shard_idx = (loader->current_shard_idx + 1) % loader->glob_result.gl_pathc;
134 loader->current_sample_idx = 0;
135 dataloader_load_shard_(loader, (int) loader->current_shard_idx);
136
137 if (loader->should_shuffle) {
138 prepare_intra_shard_indices_(loader);
139 }
140}

142void dataloader_init(DataLoader *loader,
143 const char* filename_pattern,
144 size_t B,
145 size_t T,
146 int process_rank,
147 int num_processes,
148 int should_shuffle) {
149 loader->process_rank = process_rank;
150 loader->num_processes = num_processes;
151 loader->B = B;
152 loader->T = T;
153 loader->tokens_file = NULL;
154 loader->should_shuffle = should_shuffle;
155 loader->header_bytes = HEADER_SIZE * sizeof(int);
156 loader->total_batch_size_bytes = ((loader->num_processes * (loader->B * loader->T)) * sizeof(uint16_t));
157 loader->local_batch_offset_bytes = loader->process_rank * loader->B * loader->T * sizeof(uint16_t);
158
159 // glob to get the list of files matching the pattern, these are our data shards
160 int glob_status = glob(filename_pattern, 0, NULL, &loader->glob_result);
161 if (glob_status != 0) {
162 printf("Error: failed to glob pattern: %s\n", filename_pattern);
163 exit(EXIT_FAILURE);
164 }
165 if (loader->glob_result.gl_pathc == 0) {
166 printf("Error: no files found matching the pattern: %s\n", filename_pattern);
167 exit(EXIT_FAILURE);
168 }
169
170 if (should_shuffle) {
171 mt19937_state shuffle_rng;
172 manual_seed(&shuffle_rng, 42 + process_rank);
173 loader->shuffle_rng = shuffle_rng;
174 loader->shard_indices = (int*)mallocCheck(loader->glob_result.gl_pathc * sizeof(int));
175 init_identity_permutation(loader->shard_indices, (int) loader->glob_result.gl_pathc);
176 loader->intra_shard_indices = NULL; // dynamically allocated allowing different shard sizes
177 }
178
179 // inspect and validate all shards so we don't get any runtime errors later
180 // if too slow / too many shards, may wish to revisit later
181 int64_t ntok_total = 0;
182 for (int shard_index = 0; shard_index < loader->glob_result.gl_pathc; shard_index++) {
183 int64_t shard_ntok = dataloader_load_shard_(loader, shard_index);
184 // we need at least one batch/shard, the way things are written right now.
185 // can be relaxed a lot later.
186 assert(shard_ntok >= (int64_t) (num_processes * B * T + 1));
187 ntok_total += shard_ntok;
188 }
189 // debugging prints
190 // printf("DataLoader: filename_pattern: %s\n", filename_pattern);
191 // printf("DataLoader: Found %ld tokens across %zu shards\n", ntok_total, loader->glob_result.gl_pathc);
192
193 // allocate all the space we'll need
194 loader->buffer = (uint16_t*)mallocCheck((B * T + 1) * sizeof(uint16_t));
195 loader->inputs = (int*)mallocCheck(B * T * sizeof(int));
196 loader->targets = (int*)mallocCheck(B * T * sizeof(int));
197 loader->num_tokens = ntok_total;
198
199 // reset the loader, to initialize it
200 dataloader_reset(loader);
201}

203void dataloader_load_batch(DataLoader* loader) {
204 assert(!loader->should_shuffle || (loader->should_shuffle && loader->intra_shard_indices != NULL));
205 assert(loader->current_sample_idx < loader->shard_num_samples);
206 size_t idx = loader->should_shuffle ? loader->intra_shard_indices[loader->current_sample_idx] : loader->current_sample_idx;
207 size_t global_batch_offset_bytes = idx * loader->total_batch_size_bytes;
208 int64_t current_offset = loader->header_bytes + global_batch_offset_bytes + loader->local_batch_offset_bytes;
209
210 size_t B = loader->B;
211 size_t T = loader->T;
212 // read B*T+1 uint16_t tokens from the file into buffer
213 fseekCheck(loader->tokens_file, (int) current_offset, SEEK_SET);
214 freadCheck(loader->buffer, sizeof(uint16_t), B*T+1, loader->tokens_file);
215 // decode the buffer into inputs and targets (cast to int)
216 for (int i = 0; i < B*T; i++) {
217 loader->inputs[i] = (int)loader->buffer[i];
218 loader->targets[i] = (int)loader->buffer[i+1];
219 }
220}

251// ----------------------------------------------------------------------------
252// Distributed Eval Loader
253// Many evals (like) HellaSwag and MMLU are multiple-choice
254// where there are 4 possible continuations and a label for the correct one
255// We want to load and serve these style of evals
256/*
257Copy pasting the section on the eval datafile format, from data_common.py:
258- First comes a header with 256 int32s
259- The examples follow, each example is a stream of uint16_t:
260 - <START_EXAMPLE> delimiter of 2**16-1, i.e. 65,535
261 - <EXAMPLE_BYTES>, bytes encoding this example, allowing efficient skip to next
262 - <EXAMPLE_INDEX>, the index of the example in the dataset
263 - <LABEL>, the index of the correct completion
264 - <NUM_COMPLETIONS>, indicating the number of completions (usually 4)
265 - <NUM><CONTEXT_TOKENS>, where <NUM> is the number of tokens in the context
266 - <NUM><COMPLETION_TOKENS>, repeated NUM_COMPLETIONS times
267*/
268
269// for now, could relax later
270#define ASSUMED_NUM_COMPLETIONS 4
271// helper macro for ceildiv
272#define CEIL_DIV(M, N) (((M) + (N)-1) / (N))
273
274typedef struct {
275 // variables related to distributed training
276 // each process/worker has to access different parts of the data
277 int process_rank;
278 int num_processes;
279 // hyperparameters. use size_t to prevent overflow
280 size_t B; // (micro) batch size dimension of the tensor that feeds into the model
281 size_t T; // maximum context length of the model
282 // input handling and its state
283 FILE* eval_file;
284 uint16_t* buffer; // we fread data from file into this buffer
285 // public variables that could be accessed from outside
286 int num_examples; // in total across all processes
287 int num_batches; // to process the entire dataset across all processes
288 int start_example_index; // the assignment of work for this process, start
289 int end_example_index; // and end. start is inclusive, end is exclusive
290 int current_example_index; // the next example we would read
291 int* inputs; // input tokens into transformer
292 int* targets; // target tokens for the transformer
293 char* mask; // mask=1 at all completion token locations
294 int* label; // the correct completion labels
295 int num_completions; // number of completions for this example
296} EvalLoader;

298void evalloader_reset(EvalLoader *loader) {
299 // we have to be careful that each process starts at the correct offset.
300 // For example if there are N examples in the file and 4 processes,
301 // then process 0 should start at 0, process 1 at N/4, process 2 at N/2, etc.
302 // determine how much work there is for all processes
303 int examples_per_process = CEIL_DIV(loader->num_examples, loader->num_processes);
304 int can_fit_examples = (int) (loader->B / ASSUMED_NUM_COMPLETIONS);
305 if (can_fit_examples == 0) {
306 // this could be fixed in the future, but for now keeping it simple and throw error when B too low
307 printf("HellaSwag EvalLoader: batch size %zu is < %d\n", loader->B, ASSUMED_NUM_COMPLETIONS);
308 printf("---> HINT: Disable HellaSwag eval with -h 0, or increase batch size with -b\n");
309 exit(EXIT_FAILURE);
310 }
311 loader->num_batches = CEIL_DIV(examples_per_process, can_fit_examples);
312 // determine the start and end example indices for this process
313 loader->start_example_index = examples_per_process * loader->process_rank;
314 loader->end_example_index = examples_per_process * (loader->process_rank + 1);
315 // crop the end example index to the total number of examples
316 if (loader->end_example_index > loader->num_examples) {
317 loader->end_example_index = loader->num_examples;
318 }
319 // now seek through the file to the start of that example
320 // utilize <EXAMPLE_BYTES> for efficiency
321 int64_t header_bytes = HEADER_SIZE * sizeof(int);
322 fseekCheck(loader->eval_file, (int) header_bytes, SEEK_SET);
323 for (int i = 0; i < loader->start_example_index; i++) {
324 uint16_t example_header[3];
325 // read 3 uint16_t values: <START_EXAMPLE>, <EXAMPLE_BYTES>, <EXAMPLE_INDEX>
326 freadCheck(&example_header[0], sizeof(uint16_t), 3, loader->eval_file);
327 // validate the <START_EXAMPLE> delimiter
328 assert(example_header[0] == 65535); // <START_EXAMPLE> delimiter
329 // validate the <EXAMPLE_INDEX>
330 assert(example_header[2] == i); // <EXAMPLE_INDEX> should match the loop index
331 // skip to the next example, keeping in mind that we already read the header
332 size_t remaining_bytes = example_header[1] - sizeof(uint16_t) * 3;
333 assert(remaining_bytes > 0); // we expect some bytes in the example
334 fseekCheck(loader->eval_file, (int) remaining_bytes, SEEK_CUR);
335 }
336 // now we are at the start of the example we want to start at, pointing at <START_EXAMPLE>
337 loader->current_example_index = loader->start_example_index;
338}

380void evalloader_next_example_(EvalLoader *loader, int example_batch_index) {
381 // this function populates the inputs, targets, mask, and label fields for one example
382 // because every (B,T) tensor can fit multiple examples and we want to take advantage,
383 // we also pass in the example_batch_index to indicate which example in the batch we are loading
384 // and each example takes up ASSUMED_NUM_COMPLETIONS rows in the batch
385 size_t B = loader->B;
386 size_t T = loader->T;
387 int batch_dim_offset = example_batch_index * ASSUMED_NUM_COMPLETIONS;
388 // read the current example header
389 uint16_t example_header[3];
390 freadCheck(&example_header[0], sizeof(uint16_t), 3, loader->eval_file);
391 // validate the <START_EXAMPLE> delimiter
392 assert(example_header[0] == 65535); // <START_EXAMPLE> delimiter
393 // validate the <EXAMPLE_INDEX>
394 assert(example_header[2] == loader->current_example_index); // <EXAMPLE_INDEX> should match the loop index
395 assert(example_header[2] >= loader->start_example_index && example_header[2] < loader->end_example_index);
396 // read the rest of the example (we have space for 3 more uint16_t values in buffer, it's ok)
397 size_t example_bytes = example_header[1] - sizeof(uint16_t) * 3;
398 // read example_bytes into buffer. careful that this is actually in the units of bytes
399 freadCheck(loader->buffer, sizeof(char), example_bytes, loader->eval_file);
400 // process the example label
401 int label = (int)loader->buffer[0];
402 int can_fit_examples = (int) (loader->B / ASSUMED_NUM_COMPLETIONS);
403 assert(label >= 0 && label < ASSUMED_NUM_COMPLETIONS); // we expect the label to be in [0, 4) for right now
404 assert(example_batch_index >= 0 && example_batch_index < can_fit_examples);
405 loader->label[example_batch_index] = label; // store for output
406 // process the number of completions
407 int num_completions = (int)loader->buffer[1];
408 assert(num_completions == ASSUMED_NUM_COMPLETIONS); // we expect 4 completions for now
409 assert(batch_dim_offset + num_completions <= B); // we expect to fit in the batch
410 loader->num_completions = num_completions; // store for output
411 // process the context
412 // the context is shared for all completions, so we insert it into all data rows equally
413 int context_length = (int)loader->buffer[2];
414 uint16_t *context_tokens_start = &loader->buffer[3]; // where the tokens start
415 assert(context_length > 0 && context_length < T); // context is non-empty and up to T
416 for (int b = 0; b < num_completions; b++) {
417 for (int i = 0; i < context_length; i++) {
418 int boff = batch_dim_offset + b;
419 int tok_cur = (int)context_tokens_start[i];
420 loader->inputs[boff * T + i] = tok_cur;
421 }
422 }
423 // process the completions, insert them in their row, right after the (shared) context
424 uint16_t *completions_iter = loader->buffer + 3 + context_length;
425 for (int c = 0; c < num_completions; c++) {
426 int coff = batch_dim_offset + c;
427 int completion_length = (int)completions_iter[0];
428 uint16_t *completion_tokens_start = completions_iter + 1;
429 assert(completion_length > 0 && context_length + completion_length < T); // things fit?
430 for (int i = 0; i < completion_length; i++) {
431 int tok_cur = (int)completion_tokens_start[i];
432 // at inputs, the completions simply follow the context
433 loader->inputs[coff * T + context_length + i] = tok_cur;
434 // at targets things start to get tricky
435 // we expect the last context token to predict the first completion token
436 // and then onwards from there.
437 loader->targets[coff * T + context_length + i - 1] = tok_cur;
438 // and at these positions, we want to set mask=1, because these are the
439 // positions where we want to average the loss, in each row, to determine
440 // its overall probability of following the context.
441 loader->mask[coff * T + context_length + i - 1] = 1;
442 }
443 completions_iter += 1 + completion_length; // move to the next completion
444 }
445 // advance the current example to point to the next one we'd load
446 loader->current_example_index += 1;
447}

449void evalloader_next_batch(EvalLoader *loader) {
450 size_t B = loader->B;
451 size_t T = loader->T;
452 // init mask to zeros, no need to do it for inputs & targets, the values where the mask
453 // is set will be correctly overwritten every time.
454 memset(loader->mask, 0, B * T * sizeof(char));
455 // ok here is the problem we are solving
456 // we have a batch dimension of B, which we want to take full advantage of
457 // each example has some number of completions (usually 4)
458 // so we want to pack as many examples into rows of B as we can fit
459 int can_fit_examples = (int) (B / ASSUMED_NUM_COMPLETIONS); // how many examples can we fit in the batch?
460 for (int i = 0; i < can_fit_examples; i++) {
461 if (loader->current_example_index >= loader->end_example_index) {
462 break; // this process has exhausted its work, noop from here on
463 }
464 evalloader_next_example_(loader, i);
465 }

468int evalloader_stat_losses(EvalLoader *loader, float* losses) {
469 // compute statistics of losses (B*T) resulting from a forward pass
470 // on a batch that was constructed from EvalLoader
471 // putting this functionality here because it is tightly coupled
472 // with how we construct and represent the data batches.
473 // returns the number of correct examples in this batch.
474 int correct = 0;
475 size_t B = loader->B;
476 size_t T = loader->T;
477 // iterate the examples in this batch
478 int can_fit_examples = (int) (B / ASSUMED_NUM_COMPLETIONS);
479 for (int i = 0; i < can_fit_examples; i++) {
480 float min_loss = 0.0f;
481 int min_loss_index = -1;
482 char active = 0; // is this example active or fully empty?
483 // iterate the completions in this example
484 for (int b = 0; b < ASSUMED_NUM_COMPLETIONS; b++) {
485 int boff = i * ASSUMED_NUM_COMPLETIONS + b;
486 // evaluate the quality of this completion
487 // its quality is simply the average loss over the tokens
488 float average_loss = 0.0f;
489 int count = 0;
490 for (int t = 0; t < T; t++) {
491 char mask = loader->mask[boff * T + t];
492 if (mask == 1) {
493 active = 1;
494 average_loss += losses[boff * T + t];
495 count++;
496 }
497 }
498 if (count > 0) { average_loss /= count; }
499 if (b == 0 || average_loss < min_loss) {
500 min_loss = average_loss;
501 min_loss_index = b;
502 }
503 }
504 if (active && (min_loss_index == loader->label[i])) {
505 correct += 1;
506 }
507 }
508 return correct;
509}
