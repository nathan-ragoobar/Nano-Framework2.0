409// Careful there are a few versions of GeLU, this one is the exact one used by
410// OpenAI
411struct NewGELU {
412 using T = floatX;
413
414 static void Forward(typename TTypes<T>::ConstFlat x,
415 typename TTypes<T>::Flat y) {
416 CHECK_EQ(x.size(), y.size());
417 const float sqrt_2_over_pi = std::sqrt(M_2_PI);
418
419 // y = 0.5 * x * (1.0 + tanh[sqrt(2/pi) * (x + 0.044715 * x^3)])
420 float coeff = 0.044715f;
421 y.device(g_device) =
422 0.5 * x * (1.0 + ((sqrt_2_over_pi * (x + coeff * x * x * x)).tanh()));
423 }
424
425 static void Backward(typename TTypes<T>::ConstFlat x,
426 typename TTypes<T>::ConstFlat y_grad,
427 typename TTypes<T>::Flat x_grad) {
428 CHECK_EQ(x.size(), y_grad.size());
429 CHECK_EQ(x.size(), x_grad.size());
430
431 // dL/dx = dL/dy * dy/dx
432 // = dL/dy * [ 0.5 * (1.0 + tanh[sqrt(2/pi) * (x + 0.044715 * x^3)])
433 // + 0.5 * x * (1 - (tanh[sqrt(2/pi) * (x + 0.044715 *
434 // x^3)])^2
435 // * (sqrt(2/pi) * (1 + 0.044715 * 3 * x^2))
436 // )
437 // ]
438
439 const float sqrt_2_over_pi = std::sqrt(M_2_PI);
440 float coeff = 0.044715f;
441 auto cube = coeff * x * x * x;
442 auto tanh_arg = sqrt_2_over_pi * (x + cube);
443 auto tanh_out = tanh_arg.tanh();
444 auto dydx = 0.5f * (1.0f + tanh_out) +
445 0.5f * x * (1.0f - tanh_out * tanh_out) *
446 (sqrt_2_over_pi * (1.0f + 3.0f * coeff * x * x));
447 x_grad.device(g_device) += y_grad * dydx;
448 }
449};
