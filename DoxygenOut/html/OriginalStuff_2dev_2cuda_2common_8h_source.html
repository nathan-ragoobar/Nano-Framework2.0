23// requires all 32 threads in the warp to be active, but should work for any block size
24// uses non-dynamic shared memory so every call increases shared memory requirements by 128 bytes
25// the fact it's unique shared memory allows us to avoid an extra __syncthreads() call at the end
26// but if called inside a loop, the shared memory will be implicitly reused, so set final_sync to 1
27using reduction_func_t = float (*) (float);
28
29template<reduction_func_t warp_reduction>
30__device__ inline float blockReduce(float val, bool final_sync, float out_of_bounds) {
31 // two reductions of up to 1024 threads:
32 // 1) inside warp (shuffle), 2) cross-warp (shared memory), 3) inside warp (shuffle)
33 __shared__ float shared_val[WARP_SIZE];
34 const int lane_id = threadIdx.x % WARP_SIZE;
35 const int warp_id = threadIdx.x / WARP_SIZE;
36 const int num_warps = blockDim.x / WARP_SIZE;
37
38 float warp_val = warp_reduction(val);
39 if (lane_id == 0) { shared_val[warp_id] = warp_val; }
40 __syncthreads();
41 warp_val = (lane_id < num_warps) ? shared_val[lane_id] : out_of_bounds;
42 float block_val = warp_reduction(warp_val);
43
44 if (final_sync) {
45 __syncthreads(); // only needed in loops when effectively reusing shared memory etc.
46 }
47 return block_val;
48}

50// Helper function to call blockReduce with default arguments
51template<reduction_func_t warp_reduction>
52__device__ inline float blockReduce(float val) {
53 return blockReduce<warp_reduction>(val, false, 0.0f);
54}

102// ----------------------------------------------------------------------------
103// Packed128 data structure, which forces the compiler to use 128-bit loads/stores
104// in GPUs that support (the LDG.128 and STS.128 instructions)
105// This is a bit similar to the use of float4 in the case of 32-bit floats, but
106// supports arbitrary precision.
107
108template<class ElementType>
109struct alignas(16) Packed128 {
110 // Note: = default implicitly generates a __device__ function, but explicitly
111 // adding __device__ causes a lot of warnings.
112 Packed128() = default;
113 __device__ explicit Packed128(int4 bits) {
114 static_assert(sizeof(bits) == sizeof(payload), "Size mismatch.");
115 memcpy(&payload, &bits, sizeof(bits));
116 }
117
118 __device__ static Packed128 constant(ElementType value) {
119 Packed128 result;
120 for(int k = 0; k < size; ++k) {
121 result.payload[k] = value;
122 }
123 return result;
124 }
125
126 __device__ static Packed128 zeros() {
127 return constant(0);
128 }
129
130 __device__ static Packed128 ones() {
131 return constant(1);
132 }
133
134 __device__ ElementType& operator[](int index) {
135 return payload[index];
136 }
137 __device__ const ElementType& operator[](int index) const {
138 return payload[index];
139 }
140 __device__ int4 get_bits() const {
141 int4 bits;
142 static_assert(sizeof(bits) == sizeof(payload), "Size mismatch.");
143 memcpy(&bits, &payload, sizeof(bits));
144 return bits;
145 }
146 // e.g. sizeof(int4) is 16 (4 X 4 bytes), sizeof(bfloat16) = 2, so size = 8
147 // so in the case where ElementType = bfloat16, we store 8 elements in one Packed128
148 static constexpr const int size = sizeof(int4) / sizeof(ElementType);
149 ElementType payload[size];
150};

312 D* out_gpu = (D*)malloc(num_elements * sizeof(D));
313 cudaCheck(cudaMemcpy(out_gpu, device_result, num_elements * sizeof(D), cudaMemcpyDeviceToHost));
314 int nfaults = 0;
315#ifndef ENABLE_BF16
316 float epsilon = FLT_EPSILON;
317#else
318 float epsilon = 0.079;
319#endif
320 for (int i = 0; i < num_elements; i++) {
321 // Skip masked elements
322 if(!isfinite(cpu_reference[i]))
323 continue;
324
325 // print the first few comparisons
326 if (i < 5) {
327 printf("%f %f\n", cpu_reference[i], (T)out_gpu[i]);
328 }
329 // effective tolerance is based on expected rounding error (epsilon),
330 // plus any specified additional tolerance
331 float t_eff = tolerance + fabs(cpu_reference[i]) * epsilon;
332 // ensure correctness for all elements.
333 if (fabs(cpu_reference[i] - (T)out_gpu[i]) > t_eff) {
334 printf("Mismatch of %s at %d: CPU_ref: %f vs GPU: %f\n", name, i, cpu_reference[i], (T)out_gpu[i]);
335 nfaults ++;
336 if (nfaults >= 10) {
337 free(out_gpu);
338 exit(EXIT_FAILURE);
339 }
340 }
341 }
342
343 if (nfaults > 0) {
344 free(out_gpu);
345 exit(EXIT_FAILURE);
346 }
347
348 free(out_gpu);
349}

351template<class Kernel, class... KernelArgs>
352float benchmark_kernel(int repeats, Kernel kernel, KernelArgs&&... kernel_args) {
353 cudaEvent_t start, stop;
354 // prepare buffer to scrub L2 cache between benchmarks
355 // just memset a large dummy array, recommended by
356 // https://stackoverflow.com/questions/31429377/how-can-i-clear-flush-the-l2-cache-and-the-tlb-of-a-gpu
357 // and apparently used in nvbench.
358 int deviceIdx = 0;
359 cudaCheck(cudaSetDevice(deviceIdx));
360 cudaDeviceProp deviceProp;
361 cudaCheck(cudaGetDeviceProperties(&deviceProp, deviceIdx));
362 void* flush_buffer;
363 cudaCheck(cudaMalloc(&flush_buffer, deviceProp.l2CacheSize));
364
365 cudaCheck(cudaEventCreate(&start));
366 cudaCheck(cudaEventCreate(&stop));
367 float elapsed_time = 0.f;
368 for (int i = 0; i < repeats; i++) {
369 // clear L2
370 cudaCheck(cudaMemset(flush_buffer, 0, deviceProp.l2CacheSize));
371 // now we can start recording the timing of the kernel
372 cudaCheck(cudaEventRecord(start, nullptr));
373 kernel(std::forward<KernelArgs>(kernel_args)...);
374 cudaCheck(cudaEventRecord(stop, nullptr));
375 cudaCheck(cudaEventSynchronize(start));
376 cudaCheck(cudaEventSynchronize(stop));
377 float single_call;
378 cudaCheck(cudaEventElapsedTime(&single_call, start, stop));
379 elapsed_time += single_call;
380 }
381
382 cudaCheck(cudaFree(flush_buffer));
383
384 return elapsed_time / repeats;
385}
