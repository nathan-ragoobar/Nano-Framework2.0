192extern inline int find_max_step(const char* output_log_dir) {
193 // find the DONE file in the log dir with highest step count
194 if (output_log_dir == NULL) { return -1; }
195 DIR* dir;
196 struct dirent* entry;
197 int max_step = -1;
198 dir = opendir(output_log_dir);
199 if (dir == NULL) { return -1; }
200 while ((entry = readdir(dir)) != NULL) {
201 if (strncmp(entry->d_name, "DONE_", 5) == 0) {
202 int step = atoi(entry->d_name + 5);
203 if (step > max_step) {
204 max_step = step;
205 }
206 }
207 }
208 closedir(dir);
209 return max_step;
210}

212extern inline int ends_with_bin(const char* str) {
213 // checks if str ends with ".bin". could be generalized in the future.
214 if (str == NULL) { return 0; }
215 size_t len = strlen(str);
216 const char* suffix = ".bin";
217 size_t suffix_len = strlen(suffix);
218 if (len < suffix_len) { return 0; }

220 return suffix_matches;
