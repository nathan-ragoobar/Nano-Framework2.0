38 void Forward(typename TTypes<T>::ConstMatrix x, typename TTypes<T>::Matrix y,
39 typename TTypes<T>::Flat mean, typename TTypes<T>::Flat rstd) {
40 // x: [B, D], y: [B, D]
41 CHECK_EQ(x.dimension(1), normalized_shape_);
42 CHECK_EQ(y.dimension(1), normalized_shape_);
43 CHECK_EQ(x.dimension(0), y.dimension(0));
44 int B = x.dimension(0);
45
46 // mean: [B,], rstd: [B,]
47 CHECK_EQ(mean.size(), B);
48 CHECK_EQ(rstd.size(), B);
49
50 Eigen::array<Eigen::Index, 1> along_class = {1};
51 mean.device(g_device) = x.mean(along_class);
52
53 // x_zero_centered(B, D) = x.colwise() - m.transpose()
54 // x_zero_centered_square(B, D) = x_zero_centered.array().square()
55 // var(B,) = x_zero_centered_square.rowwise().mean()
56 // std(B,) = (var + eps).sqrt()
57 // rstd(B,) = 1.f / std;
58
59 int batch_size = x.dimension(0), num_class = x.dimension(1);
60 Eigen::array<Eigen::Index, 2> batch_by_one = {batch_size, 1};
61 Eigen::array<Eigen::Index, 2> one_by_class = {1, num_class};
62 rstd.device(g_device) =
63 ((x - mean.reshape(batch_by_one).broadcast(one_by_class))
64 .square()
65 .mean(along_class) +
66 eps_)
67 .sqrt()
68 .inverse();
69
70 // normalize: (x - mean) / std
71 // && scale: (x - mean) / std * weight
72 // && shift: (x - mean) / std * weight + bias
73
74 auto weight_1d = MakeFlat(weight_->data<T>(), normalized_shape_);
75 auto bias_1d = MakeFlat(bias_->data<T>(), normalized_shape_);
76 y.device(g_device) =
77 (x - mean.reshape(batch_by_one).broadcast(one_by_class)) *
78 rstd.reshape(batch_by_one).broadcast(one_by_class) *
79 weight_1d.reshape(one_by_class).broadcast(batch_by_one) +
80 bias_1d.reshape(one_by_class).broadcast(batch_by_one);
81 }

83 void Backward(typename TTypes<T>::ConstMatrix x,
84 typename TTypes<T>::ConstMatrix y_grad,
85 typename TTypes<T>::ConstFlat mean,
86 typename TTypes<T>::ConstFlat rstd,
87 typename TTypes<T>::Matrix x_grad) {
88 // x: [B, D], y_grad: [B, D], x_grad: [B, D]
89 CHECK_EQ(x.dimension(1), normalized_shape_);
90 CHECK_EQ(y_grad.dimension(1), normalized_shape_);
91 CHECK_EQ(x_grad.dimension(1), normalized_shape_);
92 CHECK_EQ(x.dimension(0), y_grad.dimension(0));
93 CHECK_EQ(x.dimension(0), x_grad.dimension(0));
94 int B = x.dimension(0), D = x.dimension(1);
95
96 // mean: [B,], rstd: [B,]
97 CHECK_EQ(mean.size(), B);
98 CHECK_EQ(rstd.size(), B);
99
100 int batch_size = x.dimension(0), num_class = x.dimension(1);
101 Eigen::array<Eigen::Index, 2> batch_by_one = {batch_size, 1};
102 Eigen::array<Eigen::Index, 2> one_by_class = {1, num_class};
103
104 // Lazily allocate the memory for gradients
105 weight_->LazyAllocateGradient();
106 bias_->LazyAllocateGradient();
107 auto weight_1d = MakeFlat(weight_->data<T>(), normalized_shape_);
108 auto weight_grad_1d = MakeFlat(weight_->grad<T>(), normalized_shape_);
109 auto bias_1d = MakeFlat(bias_->data<T>(), normalized_shape_);
110 auto bias_grad_1d = MakeFlat(bias_->grad<T>(), normalized_shape_);
111
112 // x_grad = dL/dy * dy/dnorm
113 // * [dnorm/dxmean * dxmean/dx
114 // + dnorm/dmean * dmean/dx
115 // + dnorm/dstd * dstd/dx
116 // ]
117
118 // Eigen::Tensor<float, 2, Eigen::RowMajor>
119 norm_->LazyAllocate(B * D);
120 dnorm_->LazyAllocate(B * D);
121 dnorm_mean_->LazyAllocate(B);
122 dnorm_norm_mean_->LazyAllocate(B);
123 auto norm_2d = norm_->matrix<T>(B, D);
124 auto dnorm_2d = dnorm_->matrix<T>(B, D);
125 auto dnorm_mean_1d = dnorm_mean_->flat<T>();
126 auto dnorm_norm_mean_1d = dnorm_norm_mean_->flat<T>();
127 norm_2d.device(g_device) =
128 (x - mean.reshape(batch_by_one).broadcast(one_by_class)) *
129 rstd.reshape(batch_by_one).broadcast(one_by_class); // [B, D]
130 dnorm_2d.device(g_device) =
131 y_grad *
132 weight_1d.reshape(one_by_class).broadcast(batch_by_one); // [B, D]
133 Eigen::array<Eigen::Index, 1> along_class = {1};
134 dnorm_mean_1d.device(g_device) = dnorm_2d.mean(along_class); // [B,]
135 dnorm_norm_mean_1d.device(g_device) =
136 (dnorm_2d * norm_2d).mean(along_class); // [B,]
137 x_grad.device(g_device) +=
138 ((dnorm_2d -
139 dnorm_mean_1d.reshape(batch_by_one).broadcast(one_by_class)) -
140 norm_2d *
141 dnorm_norm_mean_1d.reshape(batch_by_one).broadcast(one_by_class)) *
142 rstd.reshape(batch_by_one).broadcast(one_by_class);
143
144 // w_grad = dL/dy * dy/dw
145 // = dL/dy * x_norm(B,D)
146 // = \sum_i^B [y_grad(B, D) \elewise_dot x_norm(B, D)]
147
148 Eigen::array<Eigen::Index, 1> along_batch = {0};
149 weight_grad_1d.device(g_device) += (y_grad * norm_2d).sum(along_batch);
150
151 // b_grad = dL/dy * dy/db
152 // = \sum_i^(B)(y_grad(B, D))
153 // = [D,]
154
155 bias_grad_1d.device(g_device) += y_grad.sum(along_batch);
156 }
