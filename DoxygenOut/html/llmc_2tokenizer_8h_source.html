28void safe_printf(const char *piece) {
29 // the tokens are raw bytes, and we we only want to print the printable ones
30 // many bytes can be various control codes, backspace, etc.
31 if (piece == NULL) { return; }
32 if (piece[0] == '\0') { return; }
33 // handle individual byte tokens
34 // every token is asserted to be at least one byte so doing piece[1] is ok
35 if (piece[1] == '\0') {
36 unsigned char byte_val = piece[0];
37 if (!(isprint(byte_val) || isspace(byte_val))) {
38 return; // weird byte, don't print it
39 }
40 }
41 printf("%s", piece);
42}

44void tokenizer_init(Tokenizer *tokenizer, const char *filename) {
45 FILE *file = fopen(filename, "rb");
46 if (file == NULL) {
47 // try to be more helpful as we just added this feature, erase later
48 printf("---\n");
49 printf("WARNING: Failed to open the tokenizer file %s\n", filename);
50 printf("The Tokenizer is a new feature added April 14 2024.\n");
51 printf("Re-run `python train_gpt2.py` to write it\n");
52 printf("---\n");
53 tokenizer->init_ok = 0;
54 return;
55 }
56 // read in the header
57 uint32_t header[256];
58 freadCheck(header, sizeof(uint32_t), 256, file);
59 assert(header[0] == 20240328);
60 int version = header[1];
61 tokenizer->vocab_size = header[2];
62 if (version == 1) {
63 // version 1 didn't include the EOT token id
64 // so we assume it is 50256, the EOT in GPT-2
65 assert(tokenizer->vocab_size == 50257); // let's be defensive here
66 tokenizer->eot_token = 50256;
67 } else if (version == 2) {
68 tokenizer->eot_token = header[3];
69 } else {
70 fprintf(stderr, "Tokenizer model file %s has bad version: %d\n", filename, version);
71 exit(EXIT_FAILURE);
72 }
73 // read in all the tokens
74 unsigned char length;
75 tokenizer->token_table = (char **)mallocCheck(tokenizer->vocab_size * sizeof(char *));
76 for (uint32_t i = 0; i < tokenizer->vocab_size; i++) {
77 freadCheck(&length, sizeof(unsigned char), 1, file);
78 assert(length > 0); // every token should be at least one character
79 char *token_bytes = (char *)mallocCheck(length + 1);
80 freadCheck(token_bytes, sizeof(char), length, file);
81 token_bytes[length] = '\0'; // Add null terminator for printing
82 tokenizer->token_table[i] = token_bytes;
83 }
84 // cleanups
85 fcloseCheck(file);
86 tokenizer->init_ok = 1;
87}
